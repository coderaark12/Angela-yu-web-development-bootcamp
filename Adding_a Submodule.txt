# Create and initialize the submodule repository
git clone https://github.com/your-username/submodule-repo.git
cd submodule-repo
touch README.md
git add README.md
git commit -m "Initial commit"
git push

# Add the submodule to your main repository
cd /path/to/main-repository
git submodule add https://github.com/your-username/submodule-repo.git path/to/submodule-folder
git submodule init
git submodule update

# Commit and push changes in your main repository
git add .gitmodules path/to/submodule-folder
git commit -m "Add submodule 'submodule-repo' at path/to/submodule-folder"
git push
